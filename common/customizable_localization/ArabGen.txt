ArabNickHouse = {
	type = character
	log_loc_errors = no
	text = {
		trigger = {
			has_any_nickname = yes
		}
		localization_key = ArabicNickGen
	}
	text = {
		trigger = {
			house.var:name_founder ?= {
				save_temporary_scope_as = good_lad
			}
			NOT = {
				root = scope:good_lad
			}
			dynasty = {
				any_in_list = {
					variable = name_pack_support_dynasty
					this = scope:good_lad
				}
			}
		}
		localization_key = Supported_House_After
	}
	text = {
		fallback = yes
		localization_key = Gui_blank
	}
}
ArabicPatCheck = {
	type = character
	text = {
		trigger = {
			culture = {
				has_cultural_pillar = language_arabic
			}
			OR = {
				has_nickname = nick_the_bastard
				has_nickname = nick_the_child_of_devilgod
				has_trait = bastard
				has_trait = wild_oat
				has_trait = disputed_heritage
			}
		}
		localization_key = arabic_nick_pat
	}
	text = {
		trigger = {
            father ?= {
                OR = {
                    AND = {
                        dynasty ?= {
                            any_in_list = {
                                variable = name_pack_support_dynasty
                                this = root.father
                            }
                        }
                    }
                    any_in_global_list = {
                        variable = name_pack_support_global
                        this = root.father
                    }     
        }
            }
		}
		localization_key = Supported_Patronym_After
	}
    text = {
        trigger = {
            exists = father
        }
        localization_key = Nonsupported_Patronym_After
    }
	text = {
		fallback = yes
		localization_key = Gui_blank
	}
}
