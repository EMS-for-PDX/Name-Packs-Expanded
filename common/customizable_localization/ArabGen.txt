	type = character
	log_loc_errors = no
	text = {
		trigger = {
			is_male = yes
			OR = {
				AND = {
					exists = dynasty
					dynasty = {
						any_in_list = {
							variable = Got_Arab_Nick_dynasty
							this = root
						}
					}
				}
				any_in_global_list = {
					variable = Got_Arab_Nick_global
					this = root
				}
			}
		}
		localization_key = ArabicNickGenMale
	}
	text = {
		trigger = {
			is_male = no
			OR = {
				AND = {
					exists = dynasty
					dynasty = {
						any_in_list = {
							variable = Got_Arab_Nick_dynasty
							this = root
						}
					}
				}
				any_in_global_list = {
					variable = Got_Arab_Nick_global
					this = root
				}
			}
		}
		localization_key = ArabicNickGenFemale
	}
	text = {
		trigger = {
			exists = house.var:name_founder
			house.var:name_founder = {
				save_temporary_scope_as = good_lad
			}
			NOT = {
				root = scope:good_lad
			}
			dynasty = {
				any_in_list = {
					variable = name_pack_support_dynasty
					this = scope:good_lad
				}
			}
		}
		localization_key = Supported_House_After
	}
	text = {
		fallback = yes
		localization_key = Gui_blank
	}
}
