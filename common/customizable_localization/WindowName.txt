WindowName = {
	type = character
	text = {
		trigger = {
			culture = {
				is_npe_supported_culture = yes
				turkish_check = yes
			}
			has_religion = religion:islam_religion
			OR = {
				AND = {
					OR = {
						is_independent_ruler = yes
						dynasty = {
							any_in_list = {
								variable = dynasty_independent

							}

					highest_held_title_tier >= tier_kingdom
				}
				AND = {
					exists = primary_spouse.dynasty

							is_independent_ruler = yes

									variable = dynasty_independent
									this = root.primary_spouse
								}
							}
						}
						highest_held_title_tier >= tier_kingdom
					}
				}
				AND = {
					is_male = yes
					exists = father
					father = {
						OR = {
							is_independent_ruler = yes
							dynasty = {
								any_in_list = {
									variable = dynasty_independent
									this = root.father
								}
							}
						}
						highest_held_title_tier >= tier_kingdom
					}
				}
				AND = {
					is_male = yes
					exists = mother
					mother = {
						OR = {
							is_independent_ruler = yes
							dynasty = {
								any_in_list = {
									variable = dynasty_independent
									this = root.mother
								}
							}
						}

					}
				}
			}
