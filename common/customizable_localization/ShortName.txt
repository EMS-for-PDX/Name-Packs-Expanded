ShortNameNotMe = {
	log_loc_errors = no
	type = character
	text = {
		trigger = {
			culture = {
				is_npe_supported_culture = yes
				turkish_check = yes
				kipchak_check = no
				karluk_check = no
				oghur_check = no
			}
			has_religion = religion:islam_religion
			OR = {
				AND = {
					OR = {
						is_independent_ruler = yes
						dynasty = {
							any_in_list = {
								variable = dynasty_independent
								this = root
							}
						}
					}
					highest_held_title_tier >= tier_kingdom
				}
				AND = {
					exists = primary_spouse.dynasty
					primary_spouse = {
						OR = {
							is_independent_ruler = yes
							dynasty = {
								any_in_list = {
									variable = dynasty_independent
									this = root.primary_spouse
								}
							}
						}
						highest_held_title_tier >= tier_kingdom
					}
				}
				AND = {
					is_male = yes
					exists = father
					father = {
						OR = {
							is_independent_ruler = yes
							dynasty = {
								any_in_list = {
									variable = dynasty_independent
									this = root.father
								}
							}
						}
						highest_held_title_tier >= tier_kingdom
					}
				}
				AND = {
					is_male = yes
					exists = mother
					mother = {
						OR = {
							is_independent_ruler = yes
							dynasty = {
								any_in_list = {
									variable = dynasty_independent
									this = root.mother
								}
							}
						}
						highest_held_title_tier >= tier_kingdom
					}
				}
			}
		}
		localization_key = Short_Name_Turkish_Muslim_Sultan
	}
	text = {
		trigger = {
			culture = {
				is_npe_supported_culture = yes
				turkish_check = yes
			}
			has_religion = religion:islam_religion
		}
		localization_key = Short_Name_Turkish_Muslim_Low
	}
	text = {
		trigger = {
			culture = {
				is_npe_supported_culture = yes
				mongol_check = yes
			}
			OR = {
				AND = {
					OR = {
						is_independent_ruler = yes
						dynasty = {
							any_in_list = {
								variable = dynasty_independent
								this = root
							}
						}
					}
					highest_held_title_tier = tier_empire
				}
				AND = {
					NOT = {
						OR = {
							AND = {
								exists = father
								father = {
									highest_held_title_tier >= tier_kingdom
								}
							}
							AND = {
								exists = mother
								mother = {
									highest_held_title_tier >= tier_kingdom
								}
							}
						}
					}
					OR = {
						AND = {
							OR = {
								has_government = tribal_government
								has_government = feudal_government
								has_government = clan_government
							}
							OR = {
								is_independent_ruler = no
								dynasty = {
									any_in_list = {
										variable = dynasty_independent
										this = root
									}
								}
							}
							highest_held_title_tier <= tier_duchy
							highest_held_title_tier != tier_barony
						}
						AND = {
							holds_landed_title = yes
							has_government = republic_government
							highest_held_title_tier <= tier_county
						}
					}
				}
			}
		}
		localization_key = Short_Name_Mongol_Suffix
	}
	text = {
