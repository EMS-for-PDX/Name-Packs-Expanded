CharacterNameCheck = {
	type = character
	text = {
		trigger = {
			OR = {
				AND = {
                    exists = dynasty
                    dynasty = {
                        any_in_list = {
                            variable = name_pack_support_dynasty
                            this = root
                        }
                    }
                }
                any_in_global_list = {
                    variable = name_pack_support_global
                    this = root
                }
			}
		}
		localization_key = Supported_Name
	}
    text = {
        fallback = yes
		localization_key = Nonsupported_Name
    }
}
CharacterPatCheck = {
	type = character
	text = {
		trigger = {
			exists = father
            father = {
                OR = {
                    AND = {
                        exists = dynasty
                        dynasty = {
                            any_in_list = {
                                variable = name_pack_support_dynasty
                                this = root.father
                            }
                        }
                    }
                    any_in_global_list = {
                        variable = name_pack_support_global
                        this = root.father
                    }
                }
            }
		}
		localization_key = Supported_Patronym
	}
    text = {
        trigger = {
            exists = father
        }
        localization_key = Nonsupported_Patronym
    }
	text = {
		fallback = yes
		localization_key = Gui_blank
	}
}
CharacterPatCheckAfter = {
	parent = CharacterPatCheck
	suffix = "_After"
}
CharacterPatCheckBefore = {
	parent = CharacterPatCheck
	suffix = "_Before"
}
