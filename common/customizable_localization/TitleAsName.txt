TitleAsName = {
	type = character
	text = {
		trigger = {
			culture = {
				is_npe_supported_culture = yes
			}
			OR = {
				culture = {
					has_cultural_pillar = heritage_arabic
				}
				has_religion = religion:islam_religion
			}
		}
		localization_key = Title_As_Name_Arabic
	}
	text = {
		trigger = {
			culture = {
				is_npe_supported_culture = yes
				turkish_check = yes
			}
		}
		localization_key = Title_As_Name_Turkish
	}
	text = {
		trigger = {
			culture = {
				is_npe_supported_culture = yes
                armenian_check = yes
			}
		}
		localization_key = Title_As_Name_Armenian
	}
	text = {
		trigger = {
			culture = {
				is_npe_supported_culture = yes
                georgian_check = yes
			}
		}
		localization_key = Title_As_Name_Georgian
	}
	text = {
		trigger = {
			culture = {
				is_npe_supported_culture = yes
                assyrian_check = yes
			}
		}
		localization_key = Title_As_Name_Assyrian
	}
	text = {
		trigger = {
			culture = {
				is_npe_supported_culture = yes
				alan_check = yes
			}
		}
		localization_key = Title_As_Name_Alan
	}
	text = {
		trigger = {
			culture = {
				is_npe_supported_culture = yes
				has_cultural_pillar = heritage_byzantine
			}
		}
		localization_key = Title_As_Name_Greek
	}
	text = {

			culture = {

		}
		localization_key = Title_As_Name_Turkish
