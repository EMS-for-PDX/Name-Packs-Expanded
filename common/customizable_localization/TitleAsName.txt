TitleAsName = {
	type = character
	text = {
		trigger = {
			culture = {
				is_cnp_supported_culture = yes
				turkish_check = yes
			}
			has_religion = religion:islam_religion
			OR = {
				AND = {
					OR = {
						is_independent_ruler = yes
						dynasty = {
							any_in_list = {
								variable = dynasty_independent
								this = root
							}
						}
					}
					highest_held_title_tier >= tier_kingdom
				}
				AND = {
					exists = primary_spouse
					primary_spouse = {
						OR = {
							is_independent_ruler = yes
							dynasty = {
								any_in_list = {
									variable = dynasty_independent
									this = root.primary_spouse
								}
							}
						}
						highest_held_title_tier >= tier_kingdom
					}
				}
				AND = {
					is_male = yes
					exists = father
					father = {
						OR = {
							is_independent_ruler = yes
							dynasty = {
								any_in_list = {
									variable = dynasty_independent
									this = root.father
								}
							}
						}
						highest_held_title_tier >= tier_kingdom
					}
				}
				AND = {
					is_male = yes
					exists = mother
					mother = {
						OR = {
							is_independent_ruler = yes
							dynasty = {
								any_in_list = {
									variable = dynasty_independent
									this = root.mother
								}
							}
						}
						highest_held_title_tier >= tier_kingdom
					}
				}
			}
		}
		localization_key = Title_As_Name_Turkish_Muslim_Sultan
	}
	text = {
		trigger = {
			culture = {
				is_cnp_supported_culture = yes
				turkish_check = yes
			}
			has_religion = religion:islam_religion
		}
		localization_key = Title_As_Name_Turkish_Muslim_Low
	}
	text = {
		trigger = {
			culture = {
				is_cnp_supported_culture = yes
			}
			OR = {
				culture = {
					has_cultural_pillar = heritage_arabic
				}
				has_religion = religion:islam_religion
			}
		}
		localization_key = Title_As_Name_Arabic
	}
	text = {
		trigger = {
			culture = {
				is_cnp_supported_culture = yes
                armenian_check = yes
			}
		}
		localization_key = Title_As_Name_Armenian
	}
	text = {
		trigger = {
			culture = {
				is_cnp_supported_culture = yes
                georgian_check = yes
			}
		}
		localization_key = Title_As_Name_Georgian
	}
	text = {
		trigger = {
			culture = {
				is_cnp_supported_culture = yes
                assyrian_check = yes
			}
		}
		localization_key = Title_As_Name_Assyrian
	}
	text = {
		trigger = {
			culture = {
				is_cnp_supported_culture = yes
				alan_check = yes
			}
		}
		localization_key = Title_As_Name_Alan
	}
	text = {
		trigger = {
			culture = {
				is_cnp_supported_culture = yes
				has_cultural_pillar = heritage_byzantine
			}
		}
		localization_key = Title_As_Name_Greek
	}
	text = {
		trigger = {
			culture = {
				is_cnp_supported_culture = yes
				has_cultural_pillar = heritage_iranian
			}
		}
		localization_key = Title_As_Name_Persian
	}
	text = {
		trigger = {
			culture = {
				is_cnp_supported_culture = yes
				turkish_check = yes
				uyghur_check = yes
			}
			is_independent_ruler = yes
			highest_held_title_tier = tier_empire
		}
		localization_key = Title_As_Name_Uyghur
	}
	text = {
		trigger = {
			culture = {
				is_cnp_supported_culture = yes
				turkish_check = yes
				uyghur_check = yes
			}
			is_independent_ruler = yes
			highest_held_title_tier = tier_kingdom
		}
		localization_key = Title_As_Name_Uyghur
	}
	text = {
		trigger = {
			culture = {
				is_cnp_supported_culture = yes
				turkish_check = yes
				karluk_check = yes
			}
			is_independent_ruler = yes
			highest_held_title_tier >= tier_kingdom
		}
		localization_key = Title_As_Name_Turkish
	}
	text = {
		trigger = {
			culture = {
				is_cnp_supported_culture = yes
				turkish_check = yes
			}
		}
		localization_key = Title_As_Name_Turkish
	}
	text = {

		localization_key = Title_As_Name_Default
	}
