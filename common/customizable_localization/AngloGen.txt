			}
		}
        localization_key = WhiteN
	}
	text = {
		trigger = {
			house.var:name_founder ?= {

					variable = name_pack_support_dynasty
					this = scope:good_lad
				}
			}
		}
		localization_key = Supported_House
	}
	text = {
		fallback = yes
		localization_key = Gui_blank
	}
}

}
