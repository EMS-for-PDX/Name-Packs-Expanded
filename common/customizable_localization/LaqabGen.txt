PreLaqabCheck = {
	type = character
	log_loc_errors = no
	text = {
		trigger = {
			OR = {
				dynasty ?= {
					any_in_list = {
						variable = pre_laqab_dynasty
						this = root
					}
				}
                any_in_global_list = {
                    variable = pre_laqab_global
                    this = root
                }
			}
		}
		localization_key = Pre_Laqab_Gen
	}
    text = {
        fallback = yes
		localization_key = Gui_blank
    }
}
