PreLaqabCheck = {
	type = character
	log_loc_errors = no
	text = {
		trigger = {
			OR = {
				AND = {
                    exists = dynasty
                    dynasty = {
                        any_in_list = {
                            variable = pre_laqab_dynasty
                            this = root
                        }
                    }
                }
                any_in_global_list = {
                    variable = pre_laqab_global
                    this = root
                }
			}
		}
		localization_key = Pre_Laqab_Gen
	}
    text = {
        fallback = yes
		localization_key = Gui_blank
    }
}

LaqabNameCheck = {
	type = character
	log_loc_errors = no
	text = {
		trigger = {
			OR = {
				AND = {
                    exists = dynasty
                    dynasty = {
                        any_in_list = {
                            variable = regular_laqab_dynasty
                            this = root
                        }
                    }
                }
                any_in_global_list = {
                    variable = regular_laqab_global
                    this = root
                }
			}
		}
		localization_key = Reg_Laqab_Gen
	}
	text = {
		trigger = {
			OR = {
				AND = {
                    exists = dynasty
                    dynasty = {
                        any_in_list = {
                            variable = name_pack_support_dynasty
                            this = root
                        }
                    }
                }
                any_in_global_list = {
                    variable = name_pack_support_global
                    this = root
                }
			}
		}
		localization_key = Supported_Name
	}
    text = {
        fallback = yes
		localization_key = Nonsupported_Name
    }
}
