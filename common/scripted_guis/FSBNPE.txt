toggle_NPE_Full = {
	scope = character 
	effect = {
		hidden_effect = {
			if = {
				limit = {
					exists = global_var:NPEFSB_stan
				}
				remove_global_variable = NPEFSB_stan
			}
			if = {
				limit = {
					exists = global_var:NPEFSB_short
				}
				remove_global_variable = NPEFSB_short
			}
			set_global_variable = {
				name = NPEFSB_long
				value = yes
			}
		}
	}
}
