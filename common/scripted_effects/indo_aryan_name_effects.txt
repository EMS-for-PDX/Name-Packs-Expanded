indo_aryan_male_name_effect = {
	save_scope_as = thechar
	if = {
		limit = {
			exists = dynasty
		}
		dynasty = {
			add_to_variable_list = { name = name_pack_support_dynasty target = scope:thechar }
		}
	}
	else = {
		add_to_global_variable_list = { name = name_pack_support_global target = scope:thechar }
	}
	if = {
		limit = {
			exists = dynasty.var:ia_dynasty_suffix
			OR = {
				AND = {
					exists = father
					dynasty = father.dynasty
					NOT = {
						father = {
							has_character_flag = had_eldest_son
						}
					}
				}
				AND = {
					exists = mother
					dynasty = mother.dynasty
					NOT = {
						mother = {
							has_character_flag = had_eldest_son
						}
					}
				}
			}
		}
		suffix_ia_roll_male = yes
	}

		limit = {
			exists = dynasty.var:ia_dynasty_suffix
		}

			25 = {

					OR = {

									has_character_flag = had_second_son
								}
							}
						}

							exists = mother
							dynasty = mother.dynasty

								mother = {

								}
							}
						}
					}
					add = 25
				}
				modifier = {
					OR = {
						AND = {
							exists = father
							dynasty = father.dynasty
							NOT = {
								father = {
									has_character_flag = had_third_son
								}
							}
						}
						AND = {
							exists = mother
							dynasty = mother.dynasty
							NOT = {
								mother = {
									has_character_flag = had_third_son
								}
							}
						}
					}
					add = 25
				}
				modifier = {
					OR = {
						AND = {
							exists = father
							dynasty = father.dynasty
							NOT = {
								father = {
									has_character_flag = had_fourth_son
								}
							}
						}
						AND = {
							exists = mother
							dynasty = mother.dynasty
