	add_to_global_variable_list = { name = $LAQAB$ target = this }
	remove_variable = laqab_temp
}
regular_laqab_effect = {
	if = {
		limit = {
			exists = dynasty
		}
		dynasty = {
			add_to_variable_list = { name = regular_laqab_dynasty target = prev }
		}
	}
	else = {
		add_to_global_variable_list = { name = regular_laqab_global target = this }
	}
}
