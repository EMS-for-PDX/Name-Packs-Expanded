dutch_female_name_effect = {
	save_scope_as = thechar
	if = {
		limit = {
			exists = dynasty
		}
		dynasty = {
			add_to_variable_list = { name = name_pack_support_dynasty target = scope:thechar }
		}
	}
	else = {
		add_to_global_variable_list = { name = name_pack_support_global target = scope:thechar }
	}
	random_list = {
		0 = {
			modifier = {
				exists = mother.mother
				NOT = {
					has_trait = bastard
					mother = {
						has_character_flag = mother_mother_name
					}
				}
				add = 40 #freq
			}
			mother.mother = {
				save_scope_as = the_lady
			}
			if = {
				limit = {
					OR = {
						scope:the_lady = {
							exists = dynasty
							dynasty = {
								any_in_list = {
									variable = name_pack_support_dynasty
									this = scope:the_lady
								}
							}
						}
						any_in_global_list = {
							variable = name_pack_support_global
							this = scope:the_lady
						}
					}
				}
				if = {
					limit = {
						exists = dynasty
					}
					dynasty = {
						add_to_variable_list = { name = name_pack_support_dynasty target = scope:thechar }
					}
				}
				else = {
					add_to_global_variable_list = { name = name_pack_support_global target = scope:thechar }
				}
			}
			change_first_name = { template_character = scope:the_lady }
			mother = {
				add_character_flag = mother_mother_name
			}
		}
		0 = {
			modifier = {
				mother ?= {
					OR = {
						has_character_flag = mother_mother_name
						NOT = {
							exists = mother
						}
					}
					NOT = {
						has_character_flag = mother_name
					}
				}
