on_birth_child = {
	on_actions = {
		on_birth_child_ne
        on_son_mother_birth
        on_son_father_birth
		on_daughter_father_birth
		on_daughter_mother_birth
		on_player_baptismal
	}
}
on_birth_child_ne = {
    trigger = {
        culture = {
            is_npe_supported_culture = yes
        }
    }
	effect = {
		name_gen_effect = yes
	}
}
on_player_baptismal = {
	trigger = {
		has_religion = religion:christianity_religion
		any_parent = {
			is_ai = no
		}
		culture = {
			OR = {
				has_name_list = name_list_croatian
				has_name_list = name_list_bulgarian
				has_name_list = name_list_serbian
				has_name_list = name_list_bosnian
			}
		}
	}
	effect = {
		set_variable = {
			name = getting_bapd
			value = yes
		}
		if = {
			limit = {
				father ?= {
					is_ai = no
				}
			}
			father = {
				trigger_event = {
					id = npebaptismal.001
					days = { 14 30 }
				}
			}
		}
		else_if = {
			limit = {
				mother ?= {
					is_ai = no
				}
			}
			mother = {
				trigger_event = {
					id = npebaptismal.001
					days = { 14 30 }
				}
			}
		}
		else = {
			remove_variable = getting_bapd
		}
	}
}
on_son_father_birth = {
	trigger = {
		NOT = {
			has_trait = bastard
		}
		is_male = yes
		father ?= {
			is_alive = yes
			NOT = {
				has_character_flag = had_eldest_son
			}
		}
	}
	effect = {
		save_scope_as = the_child
		father = {
			add_character_flag = had_eldest_son
		}
		father.dynasty ?= {
			add_to_variable_list = { name = dynasty_list_ctp_sons target = scope:the_child }
		}
		if = {
			limit = {
				father = {
					is_lowborn = yes
				}
			}
			add_to_global_variable_list = { name = global_list_ctp_sons target = scope:the_child }
		}
	}
}
