on_dynasty_created = {
	on_actions = {
		set_dynasty
	}
}
set_dynasty = {
	trigger = {
		dynast = {
			culture = {
				is_cnp_supported_culture = yes
			}
			save_temporary_scope_as = good_lad
			OR = {
				AND = {
                    dynasty = {
                        any_in_list = {
                            variable = name_pack_support_dynasty
                            this = scope:good_lad
                        }
                    }
                }
                any_in_global_list = {
                    variable = name_pack_support_global
                    this = scope:good_lad
                }
			}
		}
	}
	effect = {
		dynast = {
			dynasty_effect_ne = yes
		}
	}
}
