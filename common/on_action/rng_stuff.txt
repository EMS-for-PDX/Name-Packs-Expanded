five_year_everyone_pulse = {
	on_actions = {
		turk_nick
		welsh_nick
	}
}
norse_nick = {
	trigger = {
		culture = {
			is_npe_supported_culture = yes
			has_cultural_pillar = heritage_north_germanic
		}
		NOT = {
			has_character_flag = got_npe_nickname
		}
	}
	effect = {
		random = {
			chance = 1
			norse_nick_name_effect = yes
		}
	}
}
welsh_nick = {
	trigger = {
		culture = {
			is_npe_supported_culture = yes
			has_cultural_pillar = heritage_brythonic
		}
		NOT = {
			has_character_flag = got_npe_nickname
		}
	}
	effect = {
		random = {
			chance = 1
			welsh_nick_name_effect = yes
		}
	}
}
turk_nick = {
	trigger = {
		culture = {
			is_npe_supported_culture = yes
			has_cultural_pillar = heritage_turkic
			OR = {
				has_cultural_pillar = language_turkic
				has_cultural_pillar = language_oghur
			}
		}
		NOT = {
			has_character_flag = got_npe_nickname
		}
	}
	effect = {
		random = {
			chance = 1
			turkish_nicks = yes
		}
	}
}
mongol_nick = {
	trigger = {
		culture = {
			is_npe_supported_culture = yes
			has_cultural_pillar = heritage_mongol
			has_cultural_pillar = language_mongol
			}
		}
		NOT = {
			has_character_flag = got_npe_nickname
		}
	}
	effect = {
		random = {
			chance = 1
			mongol_nicks = yes
		}
	}
}
